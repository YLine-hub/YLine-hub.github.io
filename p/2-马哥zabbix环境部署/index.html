注意：

Zabbix 发版规划

目前的发布计划周期为六个月，每六个月将有一个新的Zabbix稳定版本发布。

每一年版Zabbix将会发布

Zabbix LTS（长期支持版本）发布 Zabbix LTS 版本在五年内为 Zabbix 用户提供支持服务，包括三年的全面支持（基础的、紧急的以及安全性上的问题）和两年的最底限度支持（仅限紧急的和安全性上的问题）。 Zabbix 标准版本发布。 标准 Zabbix 发行版对 Zabbix 客户提供全面支持（一般支持、关键支持、安全支持）6个月，直到下一次 Zabbix 稳定版发布，此外还将客户提供限制支持（关键支持和安全支持）6个月。

#!/bin/bash

# 定义 Zabbix 版本
ZABBIX_MAJOR_VER=7.0
ZABBIX_VER=${ZABBIX_MAJOR_VER}-1

# 定义镜像源URL
URL="mirror.tuna.tsinghua.edu.cn/zabbix"

# 定义 MySQL 相关配置
MYSQL_HOST=localhost
MYSQL_ZABBIX_USER="zabbix@localhost"
MYSQL_ZABBIX_PASS='password'
MYSQL_ROOT_PASS='password'

# 定义字体文件
FONT=STKAITI.TTF

# 获取本机IP地址
ZABBIX_IP=$(hostname -I | awk '{print $1}')

# 定义颜色输出函数
GREEN="echo -e \E[32;1m"
END="\E[0m"

# 加载系统发行版信息
. /etc/os-release

# 颜色输出函数
color(){
    RES_COL=60
    MOVE_TO_COL="echo -en \\033[${RES_COL}G"
    SETCOLOR_SUCCESS="echo -en \\033[1;32m" 
    SETCOLOR_FAILURE="echo -en \\033[1;31m"
    SETCOLOR_WARNING="echo -en \\033[1;33m"
    SETCOLOR_NORMAL="echo -en \\033[1;32m"
    echo -n "$1" && $MOVE_TO_COL
    echo -n "["
    if [ "$2" = "success" -o "$2" = "0" ];then
        ${SETCOLOR_SUCCESS}
        echo -n $" OK "
    elif [ "$2" = "failure" -o "$2" = "1" ];then
        ${SETCOLOR_FAILURE}
        echo -n $"FAILED"
    else
        ${SETCOLOR_WARNING}
        echo -n $"WARNING"
    fi
    ${SETCOLOR_NORMAL}
    echo -n "]"
    echo
}

# 检查命令是否执行
check_command_status(){
    if [ $? -ne 0 ];then
        color "$1" 1
        exit 1
    fi
    color "$1" 0
}

# 检查权限
check_permissions(){
    # 如果$EUID不等于0
    if [ "$EUID" -ne 0 ];then
        echo "please run this script with root privileges."
        exit 1
    fi
}

# 安装 MySQL 数据库
install_mysql(){
    # && 左边成功（返回0）则执行右侧命令
    # 如果$MYSQL_HOST不等于localhost则退出
    [ "$MYSQL_HOST" != "localhost" ] && return
    # 如果是centos或者rocky则执行
    if [ "$ID" = "centos" -o "$ID" = "rocky" ];then
        # 获取版本
        VERSION_ID=$(echo "$VERSION_ID" | cut -d . -f1)
        if [ "$VERSION_ID" = "8" ];then
            yum -y install mysql-server
            check_command_status "install mysql-server"
            systemctl enable --now mysqld
            check_command_status "enable and start mysqld"
        elif [ "$VERSION_ID" = "7" ];then
            yum -y install mariadb-server
            check_command_status "install mariadb-server"
            systemctl enable --now mariadb
            check_command_status "enable and start mariadb"
        else
            color "unsupported operating systems ,exit" 1
            exit 1
        fi
    # 如果不是centos或者rocky（那么就是ubuntu）则执行
    else
        apt update
        check_command_status "update the apt source"
        rm -rf /var/lib/dpkg/lock*
        apt install -y mysql-server
        check_command_status "install mysql-server"
        sed -i "/^bind-address.*/c bind-address = 0.0.0.0" /etc/mysql/mysql.conf.d/mysqld.cnf
        check_command_status "update MySQL bind-address"
        systemctl restart mysql
        check_command_status "restart MySQL service"
    fi
    # 设置 MySQL 密码
    mysqladmin -uroot password "$MYSQL_ROOT_PASS"
    check_command_status "set MySQL root password"
    mysql -uroot -p"$MYSQL_ROOT_PASS" <<EOF
set global log_bin_trust_function_creators = 0;
create database zabbix character set utf8mb4 collate utf8mb4_bin;
create user $MYSQL_ZABBIX_USER identified by "$MYSQL_ZABBIX_PASS";
GRANT SUPER ON *.* TO $MYSQL_ZABBIX_USER;
grant all privileges on zabbix.* to $MYSQL_ZABBIX_USER;
set global log_bin_trust_function_creators = 1;
quit
EOF
    check_command_status "configure the MySQL database and users"
}

# 安装 zabbix
install_zabbix(){
    if [ "$ID" = "centos" -o "$ID" = "rocky" ];then
        rpm -Uvh https://${URL}/zabbix/${ZABBIX_MAJOR_VER}/rhel/${VERSION_ID}/x86_64/zabbix-release-${ZABBIX_VER}.el${VERSION_ID}.noarch.rpm
        check_command_status "install zabbix yum repository"
        sed -i "s#repo.zabbix.com#${URL}#" /etc/yum.repos.d/zabbix.repo
        check_command_status "update zabbix yum repository"
        VERSION_ID=$(echo "$VERSION_ID" | cut -d . -f1)
        if [ "$VERSION_ID" = "8" ];then
            yum -y install zabbix-server-mysql zabbix-web-mysql zabbix-apache-conf zabbix-sql-scripts zabbix-selinux-policy zabbix-agent zabbix-get langpacks-zh_CN
            check_command_status "install zabbix related software packages (CentOS 8)"
        else
            yum -y install zabbix-server-mysql zabbix-agent zabbix-get
            check_command_status "install zabbix basic software packages (CentOS 7)"
            yum -y install centos-release-scl
            check_command_status "install CentOS SCL repository"
            rpm -q yum-utils || yum -y install yum-utils
            check_command_status "install yum-utils"
            yum-config-manger --enable zabbix-frontend
            check_command_stauts "enable zabbix-frontend repository"
            yum -y install zabbix-web-mysql-scl zabbix-apache-conf-scl
            check_command-status "install zabbix-frotend software packages (CentOS 7)"
        fi
    else
        wget https://${URL}/zabbix/${ZABBIX_MAJOR_VER}/ubuntu/pool/main/z/zabbix-release/zabbix-release_latest_${ZABBIX_MAJOR_VER}+${ID}${VERSION_ID}_all.deb
        check_command_status "wget zabbix packages deb"
        rm -rf /etc/apt/sources.list.d/*
        cat > /etc/apt/sources.list <<-eof
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ noble main restricted universe multiverse
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ noble-updates main restricted universe multiverse
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ noble-backports main restricted universe multiverse
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ noble-security main restricted universe multiverse
eof
        apt update
        check_command_status "update zabbix apt repository"
        dpkg -i zabbix-release_latest_${ZABBIX_MAJOR_VER}+${ID}${VERSION_ID}_all.deb
        apt update
        check_command_status "install zabbix deb"
        apt install -y zabbix-server-mysql zabbix-frontend-php zabbix-apache-conf zabbix-sql-scripts zabbix-agent language-pack-zh-hans
    fi
    zcat /usr/share/zabbix-sql-scripts/mysql/server.sql.gz | mysql --default-character-set=utf8mb4 -uzabbix -p${MYSQL_ZABBIX_PASS} zabbix 
    check_command_status "init schema and data"  
    # [#]* 表示 0 个或多个#号
    # \s* 表示 0 个或多个空格
    sed -i "/^[#]*\s*DBPassword/c DBPassword=${MYSQL_ROOT_PASS}"  /etc/zabbix/zabbix_server.conf
    check_command_status "set DBPassword"
#    sed -i '$a\php_value[date.timezone] = Asia/Shanghai' /etc/zabbix/php-fpm.conf
#    check_command_stauts "set timezone"
    systemctl restart zabbix-server zabbix-agent apache2
    systemctl enable zabbix-server zabbix-agent apache2
    check_command_stauts "enable and start zabbix-server"
}

set_zabbix_chinese_font(){
    if [ -f ./STKAITI.TTF ];then
        mv ./STKAITI.TTF /usr/share/zabbix/assets/fonts/graphfont.ttf
        check_command_status "set zabbix chinese font"
    fi
    exit 1
}

main(){
    check_permissions
    install_mysql
    install_zabbix
    set_zabbix_chinese_font
}

main

颜色代码说明 

1

echo -en "\\033[1;32m"

  • \\033[1;32m ANSI 颜色转义序列

    • \\033 ESC 字符 （八进制033）
    • [ 控制序列开始
    • 1 样式代码
    • ; 参数分隔符
    • 32 颜色代码
    • m SGR 命令结束（设置图形显示属性）

  • 颜色代码：

    • 30 黑色
    • 31 红色
    • 32 绿色
    • 33 黄色
    • 34 蓝色
    • 35 洋红色
    • 36 青色
    • 37 白色

  • 样式代码：

    • 0 重置所有属性
    • 1 粗体/高亮
    • 2 暗色
    • 3 斜体
    • 4 下划线
    • 5 闪烁
    • 7 反显
    • 8 隐藏
