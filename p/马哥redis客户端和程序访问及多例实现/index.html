#!/bin/bash
NUM=10000
PASS=123456
HOST=127.0.0.1
PORT=6379
DATABASE=0

for i in `seq $NUM`;do
    redis-cli -h ${HOST} -a "$PASS" -p ${PORT} -n ${DATABASE} --no-auth-warning  set key${i} value${i}
    echo "key${i} value${i} 写入完成"
done

echo "$NUM个 key写入到Redis完成"

[root@ubuntu2204 redis-go]#cat main.go 
package main

import (
    "context"
    "fmt"
    "github.com/redis/go-redis/v9"
)

var ctx = context.Background()
func main() {
    rdb := redis.NewClient(&redis.Options{
        Addr:     "127.0.0.1:6379",
        Password: "123456",
        DB:       0,
    })
   
    _, err := rdb.Ping(ctx).Result()
    if err != nil {
        fmt.Printf("连接redis出错，错误信息：%v", err)
        return
    }

    for i:=1;i<=10000;i++ {
        key := fmt.Sprintf("key%d", i)
        value := fmt.Sprintf("value%d", i)
        err = rdb.Set(ctx, key, value,0).Err()
        if err != nil {
            panic(err)
        }
        
        keys, err := rdb.Keys(ctx, key).Result()
        if err != nil {
            panic(err)
        }
        fmt.Println(keys)
    }
}

go编译问题
  • 问题

编译时，go自动安装go1.24，但出现报错

1
2

go: downloading go1.24 (linux/amd64)
go: download go1.24 for linux/amd64: toolchain not available

go安装版本转换器(toolchain forwarder)，而不是真正的go二进制文件

原因：你通过包管理器安装了 golang-go，但版本太低

查看go版本

1
2
3
4

# 查看包管理器安装的版本
rpm -qa | grep golang   # CentOS/RHEL
# 或
dpkg -l | grep golang   # Ubuntu/Debian

可以看到当前go版本为1.22，但是正在安装1.24版本的

  • 解决

删除现有的所有 Go 包

1
2
3
4
5
6
7
8

# Ubuntu/Debian:
sudo apt remove --purge golang-go
sudo apt autoremove

# CentOS/RHEL:
sudo yum remove golang
# 或
sudo dnf remove golang

删除旧的 Go 安装目录（如果存在）

1

sudo rm -rf /usr/local/go

下载并安装官方 Go（使用国内镜像）

1
2

wget https://golang.google.cn/dl/go1.24.4.linux-amd64.tar.gz
sudo tar -C /usr/local -xzf go1.24.4.linux-amd64.tar.gz

配置 PATH（永久生效）

1
2

echo 'export PATH=$PATH:/usr/local/go/bin' >> ~/.bashrc
source ~/.bashrc

验证

1

go version

Redis多例实现

查看redis目录

1

tree /apps/redis

进入redis/etc目录

1

cd /apps/redis/etc

想要多例启动，只需要一个程序启动多个不同端口的配置文件即可

redis配置文件中需要改变的内容：端口号、dump.rdb文件

复制redis.conf为redis6379.conf

1

cp redis.conf redis6379.conf

修改redis6379.conf的dump.rdb文件

1
2
3

[root@redis-110 etc ]# vim redis6379.conf
...
dbfilename dump6379.rdb

查看redis6379.conf

1

grep 6379 redis6379.conf

修改6379为6380，需要加上-n，不然就会全部打印

1

sed -n 's/6379/6380/p' redis6379.conf

修改6379为6380，并查看修改后的内容

1

sed 's/6379/6380/p' redis6379.conf | grep 6380

可以看到修改后的内容中，每个都为两行

修改6379为6380并去掉p

1

sed 's/6379/6380/' redis6379.conf | grep 6380

这时候结果只有一行

修改6379为6380，并生成redis6380.conf

1

sed 's/6379/6380/' redis6379.conf > redis6380.conf

修改6379为6381，并生成redis6381.conf

1

sed 's/6379/6381/' redis6379.conf > redis6381.conf

查看服务文件

1

cat /lib/systemd/system/redis.service

复制三套配置文件

1
2
3

cp /lib/systemd/system/redis.service /lib/systemd/system/redis6379.service
cp /lib/systemd/system/redis.service /lib/systemd/system/redis6380.service
cp /lib/systemd/system/redis.service /lib/systemd/system/redis6381.service

并按照下面的逻辑修改三个服务文件的配置文件

1
2
3

[root@redis-110 etc ]# vim /lib/systemd/system/redis6379.service
...
ExecStart=/apps/redis/bin/redis-server /apps/redis/etc/redis6379.conf --supervised systemd

可以看到复制出来的所有文件权限组和所有人都是root
