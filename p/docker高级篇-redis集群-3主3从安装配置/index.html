1

# 容器名字
--name redis-node-6
# redis镜像和版本号
redis:6.0.8
# 容器卷，宿主机地址:docker内部地址
-v /data/redis/share/redis-node-6:/data
# 使用宿主机的IP和端口，默认
--net host
# 获取宿主机root用户权限
--privileged=true
# 开启redis集群
--cluster-enabled yes
# 开启持久化
--appendonly yes
# redis端口号
--port 6386

进入容器redis-node-1并为6台机器构建集群关系

进入redis-node-1 

1

docker exec -it redis-node-1 /bin/bash

构建集群 

1

redis-cli --cluster create 172.16.140.102:6381 172.16.140.102:6382 172.16.140.102:6383 172.16.140.102:6384 172.16.140.102:6385 172.16.140.102:6386 --cluster-replicas 1
